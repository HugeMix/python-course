🎯 Задача

Создайте консольное приложение на Python, которое поможет пользователю вести список дел (To-Do List).

📌 Описание проекта

Программа должна позволять:

  • Добавлять задачи
  • Просматривать список задач
  • Удалять выполненные задачи
  • Сохранять список в файл и загружать его

💻 Пример кода


import json

def load_tasks():
    try:
        with open("tasks.json", "r", encoding="utf-8") as file:
            return json.load(file)
    except FileNotFoundError:
        return []

def save_tasks(tasks):
    with open("tasks.json", "w", encoding="utf-8") as file:
        json.dump(tasks, file, ensure_ascii=False, indent=4)

def main():
    tasks = load_tasks()
    while True:
        print("\n1. Добавить задачу")
        print("2. Показать список задач")
        print("3. Удалить задачу")
        print("4. Выйти")
        choice = input("Выберите действие: ")
        
        if choice == "1":
            task = input("Введите новую задачу: ")
            tasks.append(task)
            save_tasks(tasks)
        elif choice == "2":
            print("\nВаш список дел:")
            for i, task in enumerate(tasks, 1):
                print(f"{i}. {task}")
        elif choice == "3":
            num = int(input("Введите номер задачи для удаления: ")) - 1
            if 0 <= num < len(tasks):
                tasks.pop(num)
                save_tasks(tasks)
            else:
                print("Некорректный номер!")
        elif choice == "4":
            break
        else:
            print("Неверный ввод, попробуйте снова.")

if __name__ == "__main__":
    main()
